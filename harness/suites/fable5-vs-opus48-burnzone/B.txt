export async function getUserBookings(userId: string, tenantId: string, limit = 50, offset = 0): Promise<{ bookings: Booking[]; total: number }> {
  const result = await pool.query(
    `SELECT * FROM bookings
     WHERE user_id = $1 AND tenant_id = $2
     ORDER BY created_at DESC
     LIMIT $3 OFFSET $4`,
    [userId, tenantId, limit, offset]
  );
  const countResult = await pool.query(`SELECT COUNT(*) FROM bookings WHERE user_id = $1 AND tenant_id = $2`, [userId, tenantId]);
  return { bookings: result.rows, total: Number(countResult.rows[0].count) };
}
